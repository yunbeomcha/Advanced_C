#pragma warning(disable:4996)
#include <stdio.h>
#include <string.h>
int main() {
	char str[101] = { 0 }, word[100][101] = { 0 }, temp[101] = { 0 };

	int len = 0, cnt = 0, min, index = 1;
	int n;
	int i = 0;
	int j = 0;

	//입력받고 문자열 길이 구하기
	gets(str);
	scanf("%d", &n);
	len = strlen(str);

	//word배열에 단어저장하고 단어의 개수 cnt 구하기
	for (i = 0; i < len; i++){
		if (str[i] != ' '){
			word[cnt][j++] = str[i];
		}
		else{
			cnt++;
			j = 0;
		}
	}
	puts(word[n - 1]);
	for (int i = 0; i <= cnt; i++) {
		min = i;
		for (int j = 0; j <= cnt - i; j++) {
			if (strcmp(word[j], word[j + 1]) > 0) {
				strcpy(temp, word[j]);
				strcpy(word[j], word[j + 1]);
				strcpy(word[j + 1], temp);
			}
		}
	}
	puts(word[1]);
	return 0;
}
