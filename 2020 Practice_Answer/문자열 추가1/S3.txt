#pragma warning(disable:4996)
#include <stdio.h>
#include <string.h>


int main()
{
	char str[101] = { 0 }, word[10][101] = { 0 }, temp[101] = { 0 };
	int len = 0, cnt = 0;
	int i = 0;
	int j = 0;
	//입력받고 문자열 길이 구하기
	gets(str);
	len = strlen(str);
	//word배열에 단어저장하고 단어의 개수 cnt 구하기
	for (i = 0; i < len; i++){
		if (str[i] != ' '){
			word[cnt][j++] = str[i];
		}
		else{
			cnt++;
			j = 0;
		}
	}
	for (i = 0; i < cnt - 1; i++) {
		for (j = 0; j < cnt - i; j++) {
			if (strlen(word[j]) > strlen(word[j + 1])) {
				strcpy(temp, word[j]);
				strcpy(word[j], word[j + 1]);
				strcpy(word[j+1], temp);
			}
		}
	}
	for (i = cnt; i >= 0; i--) {
		puts(word[i]);
	}
	return 0;
}
